namespace Program
{
    public static class Solution
    {
        public static bool Success { get; set; }
        public static int Connect(int left, int right)
        {
            string l = left.ToString();
            string r = right.ToString();
            return int.Parse(l + r);
        }

        public static int Cal(int left, char op, int right)
        {
            switch (op)
            {
                case '+':
                    return left + right;
                case '-':
                    return left - right;
                case '*':
                    return left * right;
                case '/':
                    return left / right;
            }

            return -1;
        }

        public static void Dfs(List<int> num, List<char> op, int target, List<string> path)
        {
            int n = num.Count;
            int m = op.Count;
            int merge = n - m - 1;
            if (Success || merge < 0) return;
            if (n == 1)
            {
                if (m == 0)
                {
                    if (num[0] == target)
                    {
                        Success = true;
                        Console.WriteLine("success");
                        foreach (var s in path)
                        {
                            Console.WriteLine(s);
                        }
                    }
                }
                return;
            }
            //合并
            if (merge > 0)
            {
                for (int i = 0; i < n; i++)
                {
                    for (int j = 0; j < n; j++)
                    {
                        if (i == j) continue;
                        int a = num[i], b = num[j];
                        int t = Connect(a, b);
                        List<int> tmp = new List<int>(num);
                        tmp.Remove(a);
                        tmp.Remove(b);
                        string s = $"merge {a} and {b}";
                        path.Add(s);
                        tmp.Add(t);
                        Dfs(tmp, op, target, path);
                        path.RemoveAt(path.Count - 1);
                    }
                }
            }

            //运算
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    if (i == j) continue;
                    int a = num[i], b = num[j];
                    for (int k = 0; k < m; k++)
                    {
                        char c = op[k];
                        if (c == '-' && a < b) continue;
                        if (c == '/' && (a == 0 || b == 0 || a % b != 0)) continue;
                        int t = Cal(a, c, b);
                        List<int> tmpNum = new List<int>(num);
                        List<char> tmpOp = new List<char>(op);
                        tmpNum.Remove(a);
                        tmpNum.Remove(b);
                        tmpOp.RemoveAt(k);
                        tmpNum.Add(t);
                        string s = $"cal {a} {c} {b}";
                        path.Add(s);
                        Dfs(tmpNum, tmpOp, target, path);
                        path.RemoveAt(path.Count - 1);
                    }
                }
            }
        }

        public static void SolveProblem(List<int> num, List<char> op, int target)
        {
            Dfs(num, op, target, new List<string>());
        }

        public static void SolveInput()
        {
            List<int> num = new List<int>();
            List<char> op = new List<char>();
            string s1 = Console.ReadLine()!;
            var t1 = s1.Split(' ').ToList();
            foreach (var s in t1)
            {
                num.Add(int.Parse(s));
            }
            string s2 = Console.ReadLine()!;
            var t2 = s2.Split(' ').ToList();
            foreach (var s in t2)
            {
                op.Add(s[0]);
            }
            int target = int.Parse(Console.ReadLine()!);

            //检验输入
            foreach (var t in num)
            {
                Console.Write(t + " ");
            }
            Console.WriteLine();
            foreach (var t in op)
            {
                Console.Write(t + " ");
            }
            Console.WriteLine();
            Console.WriteLine(target);

            Console.ReadLine();

            SolveProblem(num, op, target);

            Console.ReadLine();
        }
        public static void Main(string[] args)
        {
            SolveInput();
        }
    }
}
